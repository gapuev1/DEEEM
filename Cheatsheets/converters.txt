using System;
using System.Globalization;
using System.Windows;
using System.Windows.Data;

namespace YourApp.Converters
{
    /// <summary>
    /// Конвертер Boolean -> Visibility
    /// </summary>
    public class BooleanToVisibilityConverter : IValueConverter
    {
        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            bool isVisible = value is bool boolValue && boolValue;
            
            // Если параметр "Inverse" - инвертируем
            if (parameter?.ToString() == "Inverse")
                isVisible = !isVisible;
                
            return isVisible ? Visibility.Visible : Visibility.Collapsed;
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            return value is Visibility visibility && visibility == Visibility.Visible;
        }
    }

    /// <summary>
    /// Конвертер для подсветки строк по скидке (для основных вариантов)
    /// </summary>
    public class DiscountToBrushConverter : IValueConverter
    {
        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            if (value is decimal discount)
            {
                // Порог скидки - меняйте под свой вариант
                // В2: >15, В3: >12, В4: >17, В5: >15
                if (discount >= 15)
                {
                    // Цвет - меняйте под свой вариант
                    // В2: #483D8B, В3: #F4A460, В4: #FFDEAD, В5: #008080
                    return new System.Windows.Media.SolidColorBrush(
                        (System.Windows.Media.Color)System.Windows.Media.ColorConverter.ConvertFromString("#483D8B"));
                }
            }
            return System.Windows.Media.Brushes.Transparent;
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            throw new NotImplementedException();
        }
    }
}
